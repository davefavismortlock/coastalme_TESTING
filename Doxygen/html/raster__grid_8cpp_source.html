60int CGeomRasterGrid::nCreateGrid(void)
61{
62 // Create the 2D CGeomCell array, NOTE this is faster than using 2D STL vectors
63 int
64 nXMax = m_pSim->nGetGridXMax(),
65 nYMax = m_pSim->nGetGridYMax();
66
67 // TODO Check if we don't have enough memory, if so return RTN_ERR_MEMALLOC
68 m_Cell = new CGeomCell * [nXMax];
69 for (int nX = 0; nX < nXMax; nX++)
70 m_Cell[nX] = new CGeomCell[nYMax];
71
72 // Initialize the CGeomCell shared pointer to the CGeomRasterGrid object
73 CGeomCell::m_pGrid = this;
74
75 return RTN_OK;
76}
