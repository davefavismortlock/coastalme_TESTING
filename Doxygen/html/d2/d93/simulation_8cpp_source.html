570int CSimulation::nDoSimulation(int nArg, char const* pcArgv[])
571{
572#ifdef RANDCHECK
573 CheckRand();
574 return RTN_OK;
575#endif
576
577 // ================================================== initialization section ================================================
578 // Hello, World!
579 AnnounceStart();
580
581 // Start the clock ticking
582 StartClock();
583
584 // Find out the folder in which the CoastalME executable sits, in order to open the .ini file (they are assumed to be in the same folder)
585 if (! bFindExeDir(pcArgv[0]))
586 return (RTN_ERR_CMEDIR);
587
588 // Deal with command-line parameters
589 int nRet = nHandleCommandLineParams (nArg, pcArgv);
590 if (nRet != RTN_OK)
591 return (nRet);
592
593 // OK, we are off, tell the user about the licence and the start time
594 AnnounceLicence();
595
596 // Read the .ini file and get the name of the run-data file, and path for output etc.
597 if (! bReadIniFile())
598 return (RTN_ERR_INI);
599
600 // We have the name of the run-data input file, so read it
601 if (! bReadRunDataFile())
602 return RTN_ERR_RUNDATA;
603
604 // Check raster GIS output format
605 if (! bCheckRasterGISOutputFormat())
606 return (RTN_ERR_RASTER_GIS_OUT_FORMAT);
607
608 // Check vector GIS output format
609 if (! bCheckVectorGISOutputFormat())
610 return (RTN_ERR_VECTOR_GIS_OUT_FORMAT);
611
612 // Open log file
613 if (! bOpenLogFile())
614 return (RTN_ERR_LOGFILE);
615
616 // Set up the time series output files
617 if (! bSetUpTSFiles())
618 return (RTN_ERR_TSFILE);
619
620 // Initialize the random number generators
621 InitRand0 (m_ulRandSeed[0]);
622 InitRand1 (m_ulRandSeed[1]);
623
624 // If we are doing Savitzky-Golay smoothing of the vector coastline(s), calculate the filter coefficients
625 if (m_nCoastSmooth == SMOOTH_SAVITZKY_GOLAY)
626 CalcSavitzkyGolayCoeffs();
627
628 // Create the raster grid object
629 m_pRasterGrid = new CGeomRasterGrid (this);
630
631 // Read in the basement layer (NOTE MUST HAVE THIS FILE), create the raster grid, then read in the basement DEM data to the array
632 AnnounceReadBasementDEM();
633 nRet = nReadRasterBasementDEM();
634 if (nRet != RTN_OK)
635 return nRet;
636
637 m_ulNumCells = m_nXGridMax * m_nYGridMax;
638
639 // Mark edge cells, as defined by the basement layer
640 nRet = nMarkBoundingBoxEdgeCells();
641 if (nRet != RTN_OK)
642 return nRet;
643
644 // // DEBUG CODE =================================================
645 // for (int n = 0; n < m_VEdgeCell.size(); n++)
646 // {
647 // LogStream << "[" << m_VEdgeCell[n].nGetX() << "][" << m_VEdgeCell[n].nGetY() << "] = {" << dGridCentroidXToExtCRSX(m_VEdgeCell[n].nGetX()) << ", " << dGridCentroidYToExtCRSY(m_VEdgeCell[n].nGetY()) << "} " << m_VEdgeCellEdge[n] << endl;
648 // }
649 // // DEBUG CODE =================================================
650
651 // If we are using the default cell spacing, then now that we know the size of the raster cells, we can set the size of profile spacing in m
652 if (bFPIsEqual(m_dCoastNormalAvgSpacing, 0.0, TOLERANCE))
653 m_dCoastNormalAvgSpacing = MIN_PROFILE_SPACING * m_dCellSide;
654 else
655 {
656 // The user specified a profile spacing, is this too small?
657 m_nCoastNormalAvgSpacing = nRound (m_dCoastNormalAvgSpacing / m_dCellSide);
658
659 if (m_nCoastNormalAvgSpacing < MIN_PROFILE_SPACING)
660 {
661 cerr << ERR << "profile spacing was specified as " << m_dCoastNormalAvgSpacing << " m, which is " << m_nCoastNormalAvgSpacing << " cells. Polygon creation works poorly if profile spacing is less than " << MIN_PROFILE_SPACING << " cells, i.e. " << MIN_PROFILE_SPACING *m_dCellSide << " m" << endl;
662
663 LogStream << ERR << "profile spacing was specified as " << m_dCoastNormalAvgSpacing << " m, which is " << m_nCoastNormalAvgSpacing << " cells. Polygon creation works poorly if profile spacing is less than " << MIN_PROFILE_SPACING << " cells, i.e. " << MIN_PROFILE_SPACING *m_dCellSide << " m" << endl;
664
665 return RTN_ERR_PROFILESPACING;
666 }
667 }
668
669 // We have at least one filename for the first layer, so add the correct number of layers. Note the the number of layers does not change during the simulation: however layers can decrease in thickness until they have zero thickness
670 AnnounceAddLayers();
671
672 for (int nX = 0; nX < m_nXGridMax; nX++)
673 for (int nY = 0; nY < m_nYGridMax; nY++)
674 m_pRasterGrid->m_Cell[nX][nY].AppendLayers (m_nLayers);
675
676 // Tell the user what is happening then read in the layer files
677 AnnounceReadRasterFiles();
678
679 for (int nLayer = 0; nLayer < m_nLayers; nLayer++)
680 {
681 // Read in the initial fine unconsolidated sediment depth file(s)
682 AnnounceReadInitialFineUnconsSedGIS (nLayer);
683 nRet = nReadRasterGISFile (FINE_UNCONS_RASTER, nLayer);
684 if (nRet != RTN_OK)
685 return (nRet);
686
687 // Read in the initial sand unconsolidated sediment depth file
688 AnnounceReadInitialSandUnconsSedGIS (nLayer);
689 nRet = nReadRasterGISFile (SAND_UNCONS_RASTER, nLayer);
690 if (nRet != RTN_OK)
691 return (nRet);
692
693 // Read in the initial coarse unconsolidated sediment depth file
694 AnnounceReadInitialCoarseUnconsSedGIS (nLayer);
695 nRet = nReadRasterGISFile (COARSE_UNCONS_RASTER, nLayer);
