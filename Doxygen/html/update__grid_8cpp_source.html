39int CSimulation::nUpdateGrid(void)
40{
41 // Go through all cells in the raster grid and calculate some this-timestep totals
42 m_dThisIterTopElevMax = -DBL_MAX;
43 m_dThisIterTopElevMin = DBL_MAX;
44 for (int nX = 0; nX < m_nXGridMax; nX++)
45 {
46 for (int nY = 0; nY < m_nYGridMax; nY++)
47 {
48 if (m_pRasterGrid->m_Cell[nX][nY].bIsCoastline())
49 m_ulThisIterNumCoastCells++;
50
51 if (m_pRasterGrid->m_Cell[nX][nY].bIsInContiguousSea())
52 {
53 // Is a sea cell
54 m_dThisIterTotSeaDepth += m_pRasterGrid->m_Cell[nX][nY].dGetSeaDepth();
55 }
56
57 double dTopElev = m_pRasterGrid->m_Cell[nX][nY].dGetOverallTopElev();
58
59 // Get highest and lowest elevations of the top surface of the DEM
60 if (dTopElev > m_dThisIterTopElevMax)
61 m_dThisIterTopElevMax = dTopElev;
62
63 if (dTopElev < m_dThisIterTopElevMin)
64 m_dThisIterTopElevMin = dTopElev;
65 }
66 }
67
68 // No sea cells?
69 if (m_ulThisIterNumSeaCells == 0)
70 // All land, assume this is an error
71 return RTN_ERR_NOSEACELLS;
72
73 // Now go through all cells again and sort out suspended sediment load
74 double dSuspPerSeaCell = m_dThisIterFineSedimentToSuspension / static_cast<double>(m_ulThisIterNumSeaCells);
75 for (int nX = 0; nX < m_nXGridMax; nX++)
76 {
77 for (int nY = 0; nY < m_nYGridMax; nY++)
78 {
79 if (m_pRasterGrid->m_Cell[nX][nY].bIsInContiguousSea())
80 m_pRasterGrid->m_Cell[nX][nY].AddSuspendedSediment(dSuspPerSeaCell);
81 }
82 }
83
84 // Go along each coastline and update the grid with landform attributes, ready for next timestep
85 for (int i = 0; i < static_cast<int>(m_VCoast.size()); i++)
86 {
87 for (int j = 0; j < m_VCoast[i].nGetCoastlineSize(); j++)
88 {
89 int nRet = nLandformToGrid(i, j);
90 if (nRet != RTN_OK)
91 return nRet;
92 }
93 }
94
95 return RTN_OK;
96}
