466{
467 if (m_VLayerAboveBasement.empty())
468 return INT_NODATA;
469
470 int nTop = static_cast<int>(m_VLayerAboveBasement.size()) - 1;

563bool CGeomCell::bIsSeaIncBeach(void) const
564{
565 if (m_bInContiguousSea)
566 // Sea
567 return true;
568
569 double
570 dWaterLevel = m_pGrid->pGetSim()->CSimulation::dGetThisIterSWL(),
571 dSedTop = m_VdAllHorizonTopElev.back();
572
573 // Beach
574 if ((m_VLayerAboveBasement.back().dGetUnconsolidatedThickness() > 0) && ((dSedTop - m_pGrid->pGetSim()->CSimulation::dGetMaxBeachElevAboveSWL()) < dWaterLevel))
575 return true;
576
577 return false;
578}

695void CGeomCell::CalcAllLayerElevsAndD50(void)
696{
697 m_VdAllHorizonTopElev.clear();
698 m_VdAllHorizonTopElev.push_back(m_dBasementElevation); // Elevation of top of the basement
699
700 // Calculate the elevation of the top of all other layers
701 int m = 0;
702 for (unsigned int n = 0; n < m_VLayerAboveBasement.size(); n++)
703 m_VdAllHorizonTopElev.push_back(m_VLayerAboveBasement[n].dGetTotalThickness() + m_VdAllHorizonTopElev[m++]); // Elevation of top of layer n
704
705 // Now calculate the d50 of the topmost unconsolidated sediment layer with non-zero thickness
706 m_dUnconsD50 = DBL_NODATA;
707 for (int n = static_cast<int>(m_VLayerAboveBasement.size()) - 1; n >= 0; n--)
708 {
709 double dUnconsThick = m_VLayerAboveBasement[n].dGetUnconsolidatedThickness();
710 if (dUnconsThick > 0)
711 {
712 // This is a layer with non-zero thickness of unconsolidated sediment
713 CRWCellSediment *pUnconsSedLayer = m_VLayerAboveBasement[n].pGetUnconsolidatedSediment();
714 double
715 dFineProp = pUnconsSedLayer->dGetFine() / dUnconsThick,
716 dSandProp = pUnconsSedLayer->dGetSand() / dUnconsThick,
717 dCoarseProp = pUnconsSedLayer->dGetCoarse() / dUnconsThick;
718
719 // Calculate d50 for the unconsolidated sediment
720 m_dUnconsD50 = (dFineProp * m_pGrid->pGetSim()->dGetD50Fine()) + (dSandProp * m_pGrid->pGetSim()->dGetD50Sand()) + (dCoarseProp * m_pGrid->pGetSim()->dGetD50Coarse());
721
722 break;
723 }
724 }
725}

728int CGeomCell::nGetLayerAtElev(double const dElev) const
729{
731 if (dElev < m_VdAllHorizonTopElev[0])
732 return ELEV_IN_BASEMENT;
733
734 for (unsigned int nLayer = 1; nLayer < m_VdAllHorizonTopElev.size(); nLayer++)
735 {
736 if ((m_VLayerAboveBasement[nLayer - 1].dGetTotalThickness() > 0) && (dElev >= m_VdAllHorizonTopElev[nLayer - 1]) && (dElev <= m_VdAllHorizonTopElev[nLayer]))
737 return (nLayer - 1);
738 }
739
740 return ELEV_ABOVE_SEDIMENT_TOP;
741}
