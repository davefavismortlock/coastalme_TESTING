131{
132 // All profiles without problems, but not start- or end-of-coast profiles
133 if ((! m_bStartOfCoast) &&
134 (! m_bEndOfCoast) &&

193bool CGeomProfile::bInsertIntersection(double const dX, double const dY, int const nSeg)
194{
195 // Safety check
196 if (nSeg >= nGetNumLineSegments())
197 return false;
198
199 vector<CGeom2DPoint>::iterator it;
200 it = m_VPoints.begin();
201
202 // Do the insertion
203 m_VPoints.insert(it + nSeg + 1, CGeom2DPoint(dX, dY));
204
205 // Now insert a line segment in the associated multi-line, this will inherit the profile/line seg details from the preceding line segment
206 CGeomMultiLine::InsertLineSegment(nSeg);
207
208 return true;
209}

261bool CGeomProfile::bIsPointInProfile(double const dX, double const dY, int &nPoint)
262{
263 CGeom2DPoint Pt(dX, dY);
264 auto it = find(m_VPoints.begin(), m_VPoints.end(), &Pt);
265 if (it != m_VPoints.end())
266 {
267 // Found, so return true and set nPoint to be the index of the point which was found
268 nPoint = static_cast<int>(it - m_VPoints.begin());
269 return true;

412int CGeomProfile::nGetCellGivenDepth(CGeomRasterGrid const *pGrid, double const dDepthIn)
413{
414 int nIndex = INT_NODATA; // If not found, i.e. if every profile cell has sea depth less than dDepthIn
415
416 for (unsigned int n = 0; n < m_VCellInProfile.size(); n++)
417 {
418 int
419 nX = m_VCellInProfile[n].nGetX(),
420 nY = m_VCellInProfile[n].nGetY();
421
422 double dCellDepth = pGrid->m_Cell[nX][nY].dGetSeaDepth();
423 if (dCellDepth >= dDepthIn)
424 {
425 nIndex = n;
426
427 if (n > 0)
428 nIndex = n - 1; // Grid CRS units
429
430 break;
431 }
432 }
433
434 if (nIndex == INT_NODATA)
435 {
436 nIndex = static_cast<int>(m_VCellInProfile.size()) - 1;
437 }
438
439 return nIndex;
440}
