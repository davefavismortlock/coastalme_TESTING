32int nNearestNeighbourIndex (vector<double> const* pVdX, double const dValue)
33{
34 double dDist = DBL_MAX;
35 int nIdx = 0;
36
37 for (unsigned int i = 0; i < pVdX->size(); ++i)
38 {
39 double dNewDist = tAbs(dValue - pVdX->at(i));
40
41 if (dNewDist <= dDist)
42 {
43 dDist = dNewDist;
44 nIdx = i;
45 }
46 }
47
48 return nIdx;
49}

56vector<double> VdInterp1(vector<double> const* pVdX, vector<double> const* pVdY, vector<double> const* pVdX_new)
57{
58 vector<double> VdY_new;
59 double dX, dY;
60 unsigned int
61 x_max_idx = static_cast<unsigned int>(pVdX->size()) - 1,
62 x_new_size = static_cast<unsigned int>(pVdX_new->size());
63
64 VdY_new.reserve(x_new_size);
65
66 for (unsigned int i = 0; i < x_new_size; ++i)
67 {
68 unsigned int idx = nNearestNeighbourIndex(pVdX, pVdX_new->at(i));
69
70 if (pVdX->at(idx) > pVdX_new->at(i))
71 {
72 if (idx > 0)
73 {
74 dX = pVdX->at(idx) - pVdX->at(idx - 1);
75 dY = pVdY->at(idx) - pVdY->at(idx - 1);
76 }
77 else
78 {
79 dX = pVdX->at(idx + 1) - pVdX->at(idx);
80 dY = pVdY->at(idx + 1) - pVdY->at(idx);
81 }
82 }
83 else
84 {
85 if (idx < x_max_idx)
86 {
87 dX = pVdX->at(idx + 1) - pVdX->at(idx);
88 dY = pVdY->at(idx + 1) - pVdY->at(idx);
89 }
90 else
91 {
92 dX = pVdX->at(idx) - pVdX->at(idx - 1);
93 dY = pVdY->at(idx) - pVdY->at(idx - 1);
94 }
95 }
96
97 double dM = dY / dX;
98 double dB = pVdY->at(idx) - pVdX->at(idx) * dM;
99
100 VdY_new.push_back(pVdX_new->at(i) * dM + dB);
101 }
102
103 return VdY_new;
104}
