46void CRWCellLayer::RemoveCliff(void)
47{
48 double
49 dNow = 0,
50 dLost = 0;
51
52 dNow = m_UnconsolidatedSediment.dGetFine();
53 dLost = m_UnconsolidatedSediment.dGetNotchFineLost();
54 m_UnconsolidatedSediment.SetFine(tMax(dNow - dLost, 0.0));
55 m_UnconsolidatedSediment.SetNotchFineLost(0);
56
57 dNow = m_UnconsolidatedSediment.dGetSand();
58 dLost = m_UnconsolidatedSediment.dGetNotchSandLost();
59 m_UnconsolidatedSediment.SetSand(tMax(dNow - dLost, 0.0));
60 m_UnconsolidatedSediment.SetNotchSandLost(0);
61
62 dNow = m_UnconsolidatedSediment.dGetCoarse();
63 dLost = m_UnconsolidatedSediment.dGetNotchCoarseLost();
64 m_UnconsolidatedSediment.SetCoarse(tMax(dNow - dLost, 0.0));
65 m_UnconsolidatedSediment.SetNotchCoarseLost(0);
66
67 dNow = m_ConsolidatedSediment.dGetFine();
68 dLost = m_ConsolidatedSediment.dGetNotchFineLost();
69 m_ConsolidatedSediment.SetFine(tMax(dNow - dLost, 0.0));
70 m_ConsolidatedSediment.SetNotchFineLost(0);
71
72 dNow = m_ConsolidatedSediment.dGetSand();
73 dLost = m_ConsolidatedSediment.dGetNotchSandLost();
74 m_ConsolidatedSediment.SetSand(tMax(dNow - dLost, 0.0));
75 m_ConsolidatedSediment.SetNotchSandLost(0);
76
77 dNow = m_ConsolidatedSediment.dGetCoarse();
78 dLost = m_ConsolidatedSediment.dGetNotchCoarseLost();
79 m_ConsolidatedSediment.SetCoarse(tMax(dNow - dLost, 0.0));
80 m_ConsolidatedSediment.SetNotchCoarseLost(0);
81}
