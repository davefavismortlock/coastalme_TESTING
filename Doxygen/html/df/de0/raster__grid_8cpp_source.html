64int CGeomRasterGrid::nCreateGrid(void)
65{
66 // Create the 2D CGeomCell array, NOTE this is faster than using 2D STL vectors
67 int
68 nXMax = m_pSim->nGetGridXMax(),
69 nYMax = m_pSim->nGetGridYMax();
70
71 // TODO Check if we don't have enough memory, if so return RTN_ERR_MEMALLOC
72 m_Cell = new CGeomCell * [nXMax];
73 for (int nX = 0; nX < nXMax; nX++)
74 m_Cell[nX] = new CGeomCell[nYMax];
75
76 // Initialize the CGeomCell shared pointer to the CGeomRasterGrid object
77 CGeomCell::m_pGrid = this;
78
79 return RTN_OK;
80}
