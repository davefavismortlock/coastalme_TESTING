41/*===============================================================================================================================
42
43Calculates both detailed and smoothed curvature for every point on a coastline
44
45===============================================================================================================================*/
46void CSimulation::DoCoastCurvature(int const nCoast, int const nHandedness)
47{
48 int nCoastSize = m_VCoast[nCoast].nGetCoastlineSize();
49
50 // Start with detailed curvature, do every point on the coastline, apart from the first and last points
51 for (int nThisCoastPoint = 1; nThisCoastPoint < (nCoastSize-1); nThisCoastPoint++)
52 {
53 // Calculate the signed curvature based on this point, and the points before and after
54 double dCurvature = dCalcCurvature(nHandedness, m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint-1), m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint), m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint+1));
55
56 // Set the detailed curvature
57 m_VCoast[nCoast].SetDetailedCurvature(nThisCoastPoint, dCurvature);
58 }
59
60 // Set the curvature for the first and last coastline points
61 double dTemp = m_VCoast[nCoast].dGetDetailedCurvature(1);
62 m_VCoast[nCoast].SetDetailedCurvature(0, dTemp);
63
64 dTemp = m_VCoast[nCoast].dGetDetailedCurvature(nCoastSize-2);
65 m_VCoast[nCoast].SetDetailedCurvature(nCoastSize-1, dTemp);
66
67 // Now create the smoothed curvature
68 int const WINDOWSIZE = 11; // TODO make this a user setting, must be odd
69 int const nHalfWindow = WINDOWSIZE / 2;
70
71 // Apply a running mean smoothing filter, with a variable window size at both ends of the line
72 for (int i = 0; i < nCoastSize; i++)
73 {
74 int nTmpWindow = 0;
75 double dWindowTot = 0;
76 for (int j = -nHalfWindow; j < WINDOWSIZE - nHalfWindow; j++)
77 {
78 // For points at both ends of the coastline, use a smaller window
79 int const k = i+j;
80
81 if ((k < 0) || (k >= nCoastSize))
82 continue;
83
84 dWindowTot += m_VCoast[nCoast].dGetDetailedCurvature(k);
85 nTmpWindow++;
86 }
87
88 m_VCoast[nCoast].SetSmoothCurvature(i, dWindowTot / static_cast<double>(nTmpWindow));
89 }
90
91 // Now calculate the mean and standard deviation of each set of curvature values
92 vector<double>* pVDetailed = m_VCoast[nCoast].pVGetDetailedCurvature();
93
94 double
95 dSum = accumulate(pVDetailed->begin(), pVDetailed->end(), 0.0),
96 dMean = dSum / static_cast<double>(pVDetailed->size());
97
98 m_VCoast[nCoast].SetDetailedCurvatureMean(dMean);
99
100 double
101 dSquareSum = inner_product(pVDetailed->begin(), pVDetailed->end(), pVDetailed->begin(), 0.0),
102 dSTD = sqrt(dSquareSum / static_cast<double>(pVDetailed->size()) - dMean * dMean);
103
104 m_VCoast[nCoast].SetDetailedCurvatureSTD(dSTD);
105
106 vector<double>* pVSmooth = m_VCoast[nCoast].pVGetSmoothCurvature();
107 dSum = accumulate(pVSmooth->begin(), pVSmooth->end(), 0.0),
108 dMean = dSum / static_cast<double>(pVSmooth->size());
109 m_VCoast[nCoast].SetSmoothCurvatureMean(dMean);
110 dSquareSum = inner_product(pVSmooth->begin(), pVSmooth->end(), pVSmooth->begin(), 0.0),
111 dSTD = sqrt(dSquareSum / static_cast<double>(pVSmooth->size()) - dMean * dMean);
112 m_VCoast[nCoast].SetSmoothCurvatureSTD(dSTD);
113
114 int
115 nMaxConvexSmoothedCoastPoint = -1,
116 nMaxConvexDetailedCoastPoint = -1;
117 double
118 dMaxConvexDetailed = DBL_MAX,
119 dMaxConvexSmoothed = DBL_MAX;
120 for (int mm = 0; mm < nCoastSize; mm++)
121 {
122// CGeom2DIPoint PtiThis = *m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(mm);
123// CGeom2DPoint PtThis = *m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(mm);
124
125// LogStream << "Timestep " << m_ulIter << " (" << strDispSimTime(m_dSimElapsed) << "): at coast point (" << mm << ") [" << PtiThis.nGetX() << "][" << PtiThis.nGetY() << "] = {" << PtThis.dGetX() << ", " << PtThis.dGetY() << "} curvature = " << m_VCoast[nCoast].dGetDetailedCurvature(mm) << " smoothed curvature = " << m_VCoast[nCoast].dGetSmoothCurvature(mm) << endl;
126
127 if (m_VCoast[nCoast].dGetDetailedCurvature(mm) < dMaxConvexDetailed)
128 {
129 dMaxConvexDetailed = m_VCoast[nCoast].dGetDetailedCurvature(mm);
130 nMaxConvexDetailedCoastPoint = mm;
131 }
132
133 if (m_VCoast[nCoast].dGetSmoothCurvature(mm) < dMaxConvexSmoothed)
134 {
135 dMaxConvexSmoothed = m_VCoast[nCoast].dGetSmoothCurvature(mm);
136 nMaxConvexSmoothedCoastPoint = mm;
137 }
138 }
139
140 if (bFPIsEqual(dMaxConvexDetailed, 0.0, TOLERANCE))
141 {
142 // We have a straight-line coast, so set the point of maximum detailed convexity at the coast mid-point
143 nMaxConvexDetailedCoastPoint =
144 nMaxConvexSmoothedCoastPoint = nCoastSize / 2;
145
146 m_VCoast[nCoast].SetDetailedCurvature(nMaxConvexDetailedCoastPoint, STRAIGHT_COAST_MAX_DETAILED_CURVATURE);
147
148 // And set the point of maximum smoothed convexity at the same point
149 m_VCoast[nCoast].SetSmoothCurvature(nMaxConvexSmoothedCoastPoint, STRAIGHT_COAST_MAX_SMOOTH_CURVATURE);
150 }
151
152// LogStream << "-----------------" << endl;
153// for (int kk = 0; kk < m_VCoast.back().nGetCoastlineSize(); kk++)
154// LogStream << kk << " [" << m_VCoast.back().pPtiGetCellMarkedAsCoastline(kk)->nGetX() << "][" << m_VCoast.back().pPtiGetCellMarkedAsCoastline(kk)->nGetY() << "] = {" << dGridCentroidXToExtCRSX(m_VCoast.back().pPtiGetCellMarkedAsCoastline(kk)->nGetX()) << ", " << dGridCentroidYToExtCRSY(m_VCoast.back().pPtiGetCellMarkedAsCoastline(kk)->nGetY()) << "}" << endl;
155// LogStream << "-----------------" << endl;
156
157 CGeom2DIPoint PtiMax = *m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nMaxConvexDetailedCoastPoint);
158
159 if (m_nLogFileDetail >= LOG_FILE_HIGH_DETAIL)
160 LogStream << "Timestep " << m_ulIter << " (" << strDispSimTime(m_dSimElapsed) << "): max detailed convexity (" << m_VCoast[nCoast].dGetDetailedCurvature(nMaxConvexDetailedCoastPoint) << ") at raster coastline point " << nMaxConvexDetailedCoastPoint << " [" << PtiMax.nGetX() << "][" << PtiMax.nGetY() << "] = {" << dGridCentroidXToExtCRSX(PtiMax.nGetX()) << ", " << dGridCentroidYToExtCRSY(PtiMax.nGetY()) << "}" << endl;
161
162 CGeom2DIPoint PtiMaxSmooth = *m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nMaxConvexSmoothedCoastPoint);
163 CGeom2DPoint PtMaxSmooth = *m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nMaxConvexSmoothedCoastPoint);
164
165 if (m_nLogFileDetail >= LOG_FILE_HIGH_DETAIL)
166 LogStream << "Timestep " << m_ulIter << " (" << strDispSimTime(m_dSimElapsed) << "): max smoothed convexity (" << m_VCoast[nCoast].dGetSmoothCurvature(nMaxConvexSmoothedCoastPoint) << ") near vector coastline point " << nMaxConvexSmoothedCoastPoint << ", at [" << PtiMaxSmooth.nGetX() << "][" << PtiMaxSmooth.nGetY() << "] = {" << PtMaxSmooth.dGetX() << ", " << PtMaxSmooth.dGetY() << "}" << endl;
167}

169/*===============================================================================================================================
170
171Calculates signed Menger curvature (https://en.wikipedia.org/wiki/Menger_curvature#Definition) from three points on a line. Returns +ve values for concave, -ve for convex, and zero if the points are co-linear. Curvature is multiplied by 1000 to give easier-to-read numbers
172
173===============================================================================================================================*/
174double CSimulation::dCalcCurvature(int const nHandedness, CGeom2DPoint const* pPtBefore, CGeom2DPoint const* pPtThis, CGeom2DPoint const* pPtAfter)
175{
176 double
177 dAreax4 = 2 * dTriangleAreax2(pPtBefore, pPtThis, pPtAfter),
178 dDist1 = dGetDistanceBetween(pPtBefore, pPtThis),
179 dDist2 = dGetDistanceBetween(pPtThis, pPtAfter),
180 dDist3 = dGetDistanceBetween(pPtBefore, pPtAfter);
181
182 // Safety checks
183 if (bFPIsEqual(dDist1, 0.0, TOLERANCE))
184 dDist1 = 1e-3;
185
186 if (bFPIsEqual(dDist2, 0.0, TOLERANCE))
187 dDist2 = 1e-3;
188
189 if (bFPIsEqual(dDist3, 0.0, TOLERANCE))
190 dDist3 = 1e-3;
191
192 double dCurvature = dAreax4 / (dDist1 * dDist2 * dDist3);
193
194 // Reverse if left-handed
195 int nShape = (nHandedness == LEFT_HANDED ? 1 : -1);
196
197 return (dCurvature * nShape * 1000);
198}
